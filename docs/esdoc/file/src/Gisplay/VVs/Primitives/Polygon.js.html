    /**
 * Representes a Polygon/MultiPolygon.
 * @export
 * @class Polygon
 */
export class Polygon { //Bordered Polygon

    constructor(id, bufferTriangles, bufferBorders) {
        /**
         * The identifier of this Polygon.
         * @type {string|number}
         */
        this.id = id;
        /**
         * The vertices of this polygon. Each vertex has a longitude (x) and latitude (y).
         * @type {Array<WebGLBuffer>}
         */
        this.bufferBorders = bufferBorders;
        /**
         * The triangles are the indeces that make triangles in the vertices.
         * It's a combination of vertices that make up triangles.
         * All data is stored in a WebGLBuffer.
         * @type {Array<WebGLBuffer>}
         */
        this.bufferTriangles = bufferTriangles;
    }

    getTriangles() {
        return this.bufferTriangles;
    }

    getBorders() {
        return this.bufferBorders;
    }
}

export class Polygon2 { //Bordered Polygon

    constructor(id, triangles, borders) {
        /**
         * The identifier of this Polygon.
         * @type {string|number}
         */
        this.id = id;
        /**
         * The vertices of this polygon. Each vertex has a longitude (x) and latitude (y).
         * @type {{numElements: number, buffer: WebGLBuffer}}
         */
        this.triangles = triangles;
        /**
         * The triangles are the indeces that make triangles in the vertices.
         * It's a combination of vertices that make up triangles.
         * All data is stored in a WebGLBuffer.
         * @type {{numElements: number, buffer: WebGLBuffer}}
         */
        this.borders = borders;
    }

    getInsideBuffer(){
        return this.triangles.buffer;
    }

    getInsideElements(){
        return this.triangles.numElements;
    }

    getOutsideBuffer(){
        return this.borders.buffer;
    }

    getOutsideElements(){
        return this.borders.numElements;
    }

    getTriangles() {
        return this.triangles;
    }

    getBorders() {
        return this.borders;
    }
}
