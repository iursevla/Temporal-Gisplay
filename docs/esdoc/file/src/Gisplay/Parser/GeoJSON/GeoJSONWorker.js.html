let worker = self;
/**
 * A geoJSON worker that processes polygons using earcut library (for triangulation).
 * @deprecated Unfortunately in this case it will take more time with workers than without them.
 * @class GeoJSONWorker
 */
class GeoJSONWorker {
    constructor() {
        this.polygons = [];
        this.webglContext;
        worker.onmessage = (e) => { this.onMessage(e); };
    }

    onMessage(e) {
        if (e.data.shutdown) {
            console.log("END");
            self.postMessage(this.polygons);
            this.polygons = [];
        } else {
            if (e.data.webglContext)
                for (const row of e.data)
                    this.polygons.push({ id: row.id, data: this.processCoordinates(row.coordinates, row.mp) });
        }
    }

    processCoordinates(coordinates, isMultiPolygon) {
        let polyarray = [];
        if (isMultiPolygon) { //See http://geojson.org/geojson-spec.html#multipolygon
            for (const cs of coordinates) { //Each polygon
                let outsidepolygon = cs[0];
                let tempVerts = new Array();
                for (const c of outsidepolygon)
                    tempVerts.push(c[0], c[1]);
                let triangles_vert = earcut(tempVerts);
                polyarray.push({ triangles: triangles_vert, vertices: tempVerts });
            }
        } else {
            let outsidepolygon = coordinates[0]; //See: http://geojson.org/geojson-spec.html#polygon
            let tempVerts = new Array();
            for (let out = 0; out < outsidepolygon.length - 1; out++)
                tempVerts.push(outsidepolygon[out][0], outsidepolygon[out][1]);//_vertexcount += (outsidepolygon.length + 1) / 2;
            let triangles_vert = earcut(tempVerts); // _tricount += (triangles_vert.length / 3);
            polyarray.push({ triangles: triangles_vert, vertices: tempVerts });
        }
        this.polygons.push(this.bindPolygonsToWebGLContext(polyarray));
        return polyarray;
    }

    /**
    * Convert the vertices and triangulation values to borders and triangles (inner part of the polygon).
    * @param {{vertices: Array<number>, trianglesIndeces: Array<number>}} polygonArray - The polygon outside (vertices) and how to draw the inside (trianglesIndeces). 
    * @param {number} [id=0] - The identifier of the polygon
    * @returns {Polygon} - the polygon with an identifier and it's borders and triangles in multiple buffers.
    * @see http://prntscr.com/gcz27k
    * @memberof GeoJSONParser
    */
    bindPolygonsToWebGLContext(polygonArray, id = 0) {
        const gl = this.webglContext;
        const bufferBorders = []; //Buffer borders
        const bufferTriangles = []; //WebGL Buffers with  triangles

        for (const [polygonIndex, polygon] of polygonArray.entries()) {
            let verticesOfPolygon = polygon.vertices;
            let trianglesIndeces = polygon.trianglesIndeces;

            // Create Triangles for the polygon
            let temporalTriangles = [];
            for (const triangleIndex of trianglesIndeces) {
                let [lng, lat] = [verticesOfPolygon[triangleIndex * 2], verticesOfPolygon[triangleIndex * 2 + 1]];
                temporalTriangles.push(lng, lat);
            }
            bufferTriangles.push(gl.createBuffer());
            gl.bindBuffer(gl.ARRAY_BUFFER, bufferTriangles[polygonIndex]);
            gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(temporalTriangles), gl.STATIC_DRAW);

            // Create Borders for the polygon
            let temporalVertices = [];
            for (let i = 0; i < verticesOfPolygon.length; i += 2) {
                let [lng, lat] = [verticesOfPolygon[i], verticesOfPolygon[i + 1]];
                temporalVertices.push(lng, lat);
            }
            bufferBorders.push(gl.createBuffer());
            gl.bindBuffer(gl.ARRAY_BUFFER, bufferBorders[polygonIndex]);
            gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(temporalVertices), gl.STATIC_DRAW);
        }
        return [id, bufferBorders, bufferTriangles];
    }
}
new GeoJSONWorker();
